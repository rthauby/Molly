_ =   require('underscore')
fs =  require('fs')

class FileIndex
  
  constructor: ->
    
    
  list: ->
    fs.readdir "/Users/rthauby/Dropbox/Public/blog/entries", (err, files) ->
      Molly.IO.bark "O hai, I found these entries:"
      sys.puts "[Modified Date]  [Name]"
      _.each files, (file) ->
        fs.stat "/Users/rthauby/Dropbox/Public/blog/entries/#{file}", (err, stat) ->
          sys.puts "#{stat.mtime}   #{file}"
  
Molly = _.extend({}, Molly, {FileIndex: FileIndex})
